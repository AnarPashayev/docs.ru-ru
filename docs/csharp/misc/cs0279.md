## <a name="example"></a><span data-ttu-id="b2f0e-110">Пример</span><span class="sxs-lookup"><span data-stu-id="b2f0e-110">Example</span></span>

public class myTest : IEnumerable  
{  
    IEnumerator IEnumerable.GetEnumerator()  
    {  
        yield return 0;  
    }

    }  
}
