 <span data-ttu-id="5e67a-107">Этот пример иллюстрирует ряд различных операций с нулевыми типами значений, которые могут привести к этой ошибке.</span><span class="sxs-lookup"><span data-stu-id="5e67a-107">This example illustrates a number of the different operations with nullable value types that will cause this error.</span></span>  
  
```csharp  
// CS0458.cs  
using System;  
public  class Test
{  
    public static void Main()  
    {  
        int a = 5;  
        int? b = a + null;    // CS0458  
        int? qa = 15;  
        b = qa + null;        // CS0458  
        b -= null;            // CS0458  
        int? qa2 = null;  
        b = qa2 + null;       // CS0458  
        qa2 -= null;          // CS0458  
    }  
}  
```
