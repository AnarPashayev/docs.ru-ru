 <span data-ttu-id="c8f42-105">Компилятор обнаружил недопустимую инструкцию.</span><span class="sxs-lookup"><span data-stu-id="c8f42-105">The compiler found an invalid statement.</span></span>

```csharp  
// CS1044.cs

{  
   public void Dispose()  
   {  
      Console.WriteLine("Res1.Dispose()");

   {  
      using (MyClass mc1 = new MyClass(),

      {  
      }  
   }  
}  
```
