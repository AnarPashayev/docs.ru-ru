## <a name="example"></a><span data-ttu-id="4a5cc-110">Пример</span><span class="sxs-lookup"><span data-stu-id="4a5cc-110">Example</span></span>

public static class Extensions  
{  
    public static S ExtMethod(this S s)

        return s;  
    }

{  
}

    static int Main()

    }
