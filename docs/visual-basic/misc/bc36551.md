## <a name="example"></a><span data-ttu-id="e665c-108">Пример</span><span class="sxs-lookup"><span data-stu-id="e665c-108">Example</span></span>

Namespace StringUtility  
    <Extension()> _  
    Module StringExtensions  
        <Extension()> _

        End Sub  
    End Module
